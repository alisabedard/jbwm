00124 void
00125 make_new_client(Window w, ScreenInfo * s)
00126 {
00127         Client *c;
00128 
00129         /*
00130          * First a bit of interaction with the error handler due to X's
00131          * tendency to batch event notifications.  We set a global variable
00132          * to the id of the window we're initialising then do simple X call
00133          * on that window.  If an error is raised by this (and nothing else
00134          * should do so as we've grabbed the server), the error handler
00135          * resets the variable indicating the window has already disappeared,
00136          * so we stop trying to manage it.
00137          */
00138         arwm.initialising = w;
00139         XSync(arwm.X.dpy, False);
00140         /*
00141          * If 'initialising' is now set to None, that means doing the
00142          * XFetchName raised BadWindow - the window has been removed before
00143          * we got a chance to grab the server.
00144          */
00145         if(arwm.initialising == None)
00146                 return;
00147         arwm.initialising = None;
00148         c = Client_new(w, s);
00149         if(c->flags & AR_CLIENT_DONT_MANAGE)
00150                 return;
00151         XSelectInput(arwm.X.dpy, c->window, EnterWindowMask
00152                 | PropertyChangeMask  | ColormapChangeMask);
00153 #ifdef USE_SHAPE
00154         set_shape(c);
00155 #endif /* USE_SHAPE */
00156         reparent(c);
00157         set_class_attributes(c, w);
00158         map_client(c, s);
00159         /* Enable alt-dragging within window */
00160         arwm_grab_button(w, arwm.keymasks.grab, AnyButton);
00161 }
